############################################################
#   The CMake File for OpenGlAllegro.  It provides for
#   the following libraries:  Boost, GLM, GL, GLEW,
#   stdc++, freeimageplus, Allegro, assimp, X11 and pthread.
#   Created by Edward C. Eberle <eberdeed@ebedeed.net>
#   March 2020 San Diego California USA
############################################################
cmake_minimum_required(VERSION 3.12)
if(POLICY CMP0015)
  cmake_policy(SET CMP0015 NEW)
endif()
project(openglallegro)
add_executable(openglallegro openglallegro.cpp)
add_definitions(-g -fPIC -std=c++17 -pthread)
include_directories(/usr/local/include /usr/include/glm /usr/include/GL
/usr/local/include/assimp /usr/include/boost /usr/local/include/allegro5
/usr/local/include/SDL2 /usr/include/GLFW ../../datagen/assimpopengl/include 
../../datagen/include)
link_directories(/usr/lib /usr/lib/x86_64-linux-gnu /usr/local/lib)
target_link_libraries(openglallegro stdc++ GL GLEW allegro allegro_dialog allegro_acodec 
allegro_audio freeimage freeimageplus boost_filesystem boost_system assimp assimpopengl 
openglscene pthread X11)
install(TARGETS openglallegro DESTINATION /usr/bin)
